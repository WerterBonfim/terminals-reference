```

# Nova sessão
tmux

# listar sessões
[Ctrl + b] s

# Nova janela
[ctrl + b] c

# Navega para a janela enterior
[ctrl + b] p

# Navega para a proxima janela
[ctrl + b] n

# navegar entre as janelas
[Ctrl + b] 1 

# navegar para ultima janela que você estava
[Ctrl + b] l

#renomear a janela
[Ctrl + b] ,

# lista as janelas
[Ctrl + b] w

# sair, dettach
[Ctrl + b] d

# conectar
tmux attach -t [nome conexão] ou [0, 1, 2 ...]

# dividir janelas verticalmente
[Ctrl + b] %
# navegando entre abas
[Ctrl + b] ->  <-  o

# dividir verticalmente
[Ctrl + b] "
[Ctrl + b] 
[Ctrl + b] 


```
